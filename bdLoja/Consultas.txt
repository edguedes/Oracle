-- COMENTÁRIO: SINTAXE DE UMA CONSULTA SIMPLES TRAZENDO TODOS OS REGISTROS
SELECT * FROM CLIENTE;

-- CONSULTA TRAZENDO APENAS OS NOMES DOS CLIENTES
SELECT NOME FROM CLIENTE;

-- ORDENAMENTO

-- CONSULTA TRAZENDO TODOS OS REGISTRO POR ORDENADO POR NOME 
SELECT * FROM CLIENTE ORDER BY NOME;

-- CONSULTA TRAZENDO TODOS OS REGISTRO POR ORDENADO POR NOME DE FORMA DECRESCENTE
SELECT * FROM CLIENTE ORDER BY NOME DESC;

-- CONSULTA TRAZENDO TODOS OS REGISTRO POR ORDENADO POR COLUNA DE FORMA DECRESCENTE
SELECT * FROM CLIENTE ORDER BY 3 NOME DESC;

-- OPERADORES LOGICOS = > < => <= <>
-- CONSULTANDO O CLIENTE COM O idCliente  DE NUMERO 3
SELECT * FROM CLIENTE  WHERE IDCLIENTE = 3;

-- CONSULTANDO OS CLIENTE COM O idCliente  MAIOR QUE 2
SELECT * FROM CLIENTE  WHERE IDCLIENTE >= 2;

-- CONSULTANDO OS CLIENTE COM O idCliente  DIFERENTE 2
SELECT * FROM CLIENTE  WHERE IDCLIENTE >= 2;

-- CONSULTANDO OS CLIENTE COM O idCliente  IGUAL 2 OU idCliente IGUAL 4 USA-SE OPERADOR SQL OR
SELECT * FROM CLIENTE  WHERE IDCLIENTE = 2 OR IDCLIENTE = 4;


-- CONSULTA POR NOME --
SELECT * FROM CLIENTE WHERE NOME = 'LUISA BEATRIZ';

-- CONSULTANDO POR PARTE DO NOME - USA-SE OPERADOR SQL LIKE
SELECT * FROM CLIENTE WHERE CIDADE LIKE '%SÃO%';

-- CONSULTANDO POR PARTE CIDADES QUE NÃO TENHA SÃO - USA-SE OPERADOR SQL NOT LIKE
SELECT * FROM CLIENTE WHERE CIDADE NOT LIKE '%SÃO%';

-- CONSULTA POR IDCLIENTE ESTEJA ENTRE VALORES
SELECT * FROM CLIENTE WHERE IDCLIENTE BETWEEN 2 AND 5;

-- CONSULTA POR IDCLIENTE ESTEJA ENTRE VALORES QUE AS CIDADES CONTENHA 'NA' NO NOME DA CIDADE
SELECT * FROM CLIENTE WHERE IDCLIENTE BETWEEN 2 AND 5 AND CIDADE LIKE '%NA%';

-- OPERADOR IN  PARA SELECIONAR PARTES DA LISTA --
SELECT * FROM CLIENTE WHERE IDCLIENTE IN (2,3,5);

-- OPERADOR IN  PARA SELECIONAR POR CIDADES DA LISTA --
SELECT * FROM CLIENTE WHERE CIDADE IN ('JOÃO PESSOA', 'NATAL');

-- OPERADOR IN  PARA SELECIONAR POR CIDADES DA LISTA COM NOMES DOS CLIENTE QUE COMEÇAM COM 'J' --
SELECT * FROM CLIENTE WHERE CIDADE IN ('RECIFE', 'OLINDA') AND NOME LIKE ('J%');

-- OPERADOR IN  PARA SELECIONAR PARTES QUE NÃO SEJA UM DOS CLIENTES INFORMADOS --
SELECT * FROM CLIENTE WHERE IDCLIENTE IN (2,3,5);

-- SELECIONE AS CIDADES SEM REPETICAO E ORDENADO POR CIDADE --
